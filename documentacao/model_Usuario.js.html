src/model/Usuario.js

const mongoose = require('mongoose');
const UsuarioSchema = new mongoose.Schema({
    nome: String,
    email: String,
    hash: String,
    salt: String
});
module.exports = mongoose.model('Usuario', UsuarioSchema);

Visão Geral

Esse módulo exporta um modelo do Mongoose para a entidade Usuario que irá fazer a relação entre um objeto da nossa aplicação e um documento no banco de dados.
